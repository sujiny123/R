#3행 3열 행렬 생성4
## nrow= 행의수 , ncol= 열의수

##matrix 참조 [행의위치, 열의위치]

matrix(1:9, nrow=3, byrow=T)

#행렬의 이름을 부여하는 함수 dimnames
dimnamex = list(c('r1','r2','r3'), c('c1','c2','c3'))
x= matrix(1:9, nrow =3, byrow=T, dimnames = dimnamex)
x
##행렬에 요소 추출 : 인덱스[행위치, 열위치]

x[2:3,2:3]
x[c(1,3),c(1,3)]
